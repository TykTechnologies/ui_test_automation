var Wrapper = require('./Wrapper');

class Button_object extends Wrapper{
  constructor(selector) {
    super(selector);
    this.selector = selector;
  }

/**
 * Clicking Button object
 */
  click() {
    console.log(`>>> Clicking button: ${this.selector}`);
    super.click();
  }

}

module.exports = Button_object;
